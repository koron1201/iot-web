iot-web/
├───README.md                     # プロジェクト全体の概要を記述するファイル
├───.git/                         # Gitバージョン管理用の隠しディレクトリ
│
├───Backend/                      # Python FastAPIで構築されたバックエンドアプリケーション
│   ├───main.py                   # FastAPIアプリケーションのメインファイル（エントリーポイント）
│   ├───requirements.txt          # バックエンドが必要とするPythonライブラリのリスト
│   ├───core/                     # アプリケーションの核となる設定（データベース接続など）
│   └───modules/                  # APIの各機能（エンドポイント）をモジュールとして分割
│
├───docs/                         # プロジェクトの設計や仕様に関するドキュメント
│
├───Frontend/                     # ReactとTypeScriptで構築されたフロントエンドアプリケーション
│   ├───package.json              # プロジェクト情報と必要なライブラリ(npm)のリスト
│   ├───src/                      # フロントエンドのメインソースコード
│   │   ├───components/           # 再利用可能なUI部品（コンポーネント）
│   │   ├───routes/               # 各ページに対応するコンポーネント
│   │   └───main.tsx              # Reactアプリケーションのメインファイル（エントリーポイント）
│
└───venv/                         # Pythonの仮想環境ディレクトリ
