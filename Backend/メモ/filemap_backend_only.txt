Backend/
├─── .env                     # 環境変数ファイル（データベース接続情報、APIキーなどを記載）
├─── .gitignore               # Gitがバージョン管理しないファイル/フォルダを指定
├─── main.py                  # FastAPIアプリケーションのメインファイル（APIルーターの統合など）
├─── README.md                # バックエンドに関する説明書
├─── requirements.txt         # バックエンドが必要とするPythonライブラリ（依存関係）のリスト
├─── core/                    # アプリケーションの核となる設定
│    └─── database.py          # SQLAlchemyを使ったデータベース接続エンジンとセッションの設定
├─── modules/                 # APIの各機能を役割ごとに分割したモジュール群
│    ├─── calendar/            # カレンダー機能モジュール
│    ├─── chatbot/             # チャットボット機能モジュール
│    ├─── contact/             # 問い合わせ機能モジュール
│    ├─── deliverables/        # 成果物管理機能モジュール
│    ├─── news/                # ニュース機能モジュール
│    └─── research/            # 研究内容機能モジュール
└─── メモ/                    # 個人的なメモ用ディレクトリ
     └─── filemap.txt        # おそらく以前に作成されたファイルマップのコピー
